<!-- (옵션) GA4 커스텀 이벤트 바인더: CTA/FAQ/Deposit 섹션 가시성 -->
<script>
  // CTA 클릭 추적
  document.querySelectorAll('[data-gtag]').forEach(function(el){
    el.addEventListener('click', function(){
      if (typeof gtag === 'function') {
        gtag('event', el.getAttribute('data-gtag'), {
          location: el.getAttribute('data-gtag-location')||'',
          label: el.getAttribute('data-gtag-question')||''
        });
      }
    });
  });
  // FAQ 열림 추적
  document.querySelectorAll('details[data-gtag="faq_open"]').forEach(function(d){
    d.addEventListener('toggle', function(){
      if (d.open && typeof gtag === 'function') {
        gtag('event','faq_open',{question:d.getAttribute('data-gtag-question')||''});
      }
    });
  });
  // Deposit 섹션 가시 이벤트
  (function(){
    var dep = document.querySelector('#deposit');
    if (!dep || !('IntersectionObserver' in window)) return;
    var fired=false;
    new IntersectionObserver(function(entries, obs){
      entries.forEach(function(e){
        if (!fired && e.isIntersecting && typeof gtag==='function'){
          fired=true; gtag('event','deposit_view');
          obs.disconnect();
        }
      });
    }, {threshold:.4}).observe(dep);
  })();
</script>
